-- ════════════════════════════════════════════════════════════════════════ -- TRIADA · МИГРАЦИЯ 27 — ДАТЫ ЗАВЕРШЕНИЯ + ЗАПОЛНЕНИЕ ДЛЯ СТАРЫХ ЗАДАЧ -- Чтобы в дедлайнах появились статусы «в срок / раньше / позже» у УЖЕ -- выполненных задач и этапов. Объединяет колонки, триггеры и backfill. -- Безопасно применять, даже если миграции 16/26 уже выполнены (idempotent). -- -- ВАЖНО про точность: для задач, закрытых ДО внедрения учёта, точная дата -- завершения не сохранялась. Backfill берёт время последнего изменения -- записи (created_at) как наилучшее доступное приближение. Для задач, -- завершаемых ПОСЛЕ этой миграции, дата будет точной (ставится триггером). -- Выполнять в Supabase → SQL Editor. -- ════════════════════════════════════════════════════════════════════════ -- ════════════════════════════════════════════════════════════════════════ -- 1. ЗАДАЧИ: колонка + триггер -- ════════════════════════════════════════════════════════════════════════ alter table public.project_tasks add column if not exists completed_at timestamptz; create or replace function public.set_task_completed() returns trigger language plpgsql as $fn$ begin if new.status = 'done' and (tg_op = 'INSERT' or old.status is distinct from 'done') then new.completed_at := coalesce(new.completed_at, now()); elsif new.status is distinct from 'done' then new.completed_at := null; end if; return new; end $fn$; drop trigger if exists trg_task_completed on public.project_tasks; create trigger trg_task_completed before insert or update on public.project_tasks for each row execute function public.set_task_completed(); -- ════════════════════════════════════════════════════════════════════════ -- 2. ЭТАПЫ: колонка + триггер -- ════════════════════════════════════════════════════════════════════════ alter table public.project_stages add column if not exists completed_at timestamptz; create or replace function public.set_stage_completed() returns trigger language plpgsql as $fn$ begin if new.status = 'done' and (tg_op = 'INSERT' or old.status is distinct from 'done') then new.completed_at := coalesce(new.completed_at, now()); elsif new.status is distinct from 'done' then new.completed_at := null; end if; return new; end $fn$; drop trigger if exists trg_stage_completed on public.project_stages; create trigger trg_stage_completed before insert or update on public.project_stages for each row execute function public.set_stage_completed(); -- ════════════════════════════════════════════════════════════════════════ -- 3. BACKFILL: заполнить дату для уже выполненных (приближение по created_at) -- Точной даты для старых записей нет — берём created_at. -- ════════════════════════════════════════════════════════════════════════ update public.project_tasks set completed_at = coalesce(created_at, now()) where status = 'done' and completed_at is null; update public.project_stages set completed_at = coalesce(created_at, now()) where status = 'done' and completed_at is null; -- ════════════════════════════════════════════════════════════════════════ -- VERIFY (сколько выполненных получили дату завершения) -- ════════════════════════════════════════════════════════════════════════ select 'tasks' as kind, count(*) filter (where completed_at is not null) as with_date, count(*) filter (where status='done') as done_total from public.project_tasks union all select 'stages', count(*) filter (where completed_at is not null), count(*) filter (where status='done') from public.project_stages; -- ════════════════════════════════════════════════════════════════════════ -- ROLLBACK (откат только backfill — колонки/триггеры оставить) -- ════════════════════════════════════════════════════════════════════════ -- update public.project_tasks set completed_at = null where status='done'; -- update public.project_stages set completed_at = null where status='done';
