-- ===================================================================== -- TRIADA · Фаза 7B · Атомарное продление · функция renew_project() -- Заворачивает все 5 шагов продления в ОДНУ транзакцию (всё-или-ничего): -- 1) снимок текущего цикла (этапы + задачи) → project_cycles -- 2) удаление задач -- 3) сброс этапов (первый в работу, остальные в очередь) -- 4) новый период договора (upsert) -- 5) статус проекта = active -- Если любой шаг падает — откатывается весь, частичного состояния нет. -- -- БЕЗОПАСНОСТЬ: SECURITY DEFINER (обходит RLS), поэтому ВНУТРИ явная -- проверка прав — проект должен быть моего агентства и я должен иметь -- can_edit('projects'). Иначе exception 'forbidden'. my_agency()/can_edit() -- читают JWT и работают даже внутри SECURITY DEFINER. -- -- Снимок повторяет структуру клиента (cycleCardHTML это читает): -- stages: [{name,status,idx}], tasks: [{title,status,stage,assignee_id,time_spent,due_date}] -- Идемпотентно при повторном запуске (create or replace). -- ===================================================================== create or replace function public.renew_project( p_project_id uuid, p_start date, p_end date ) returns int language plpgsql security definer set search_path = public as $$ declare v_agency uuid; v_next int; v_snap jsonb; v_old_start date; v_old_end date; v_first uuid; begin -- ---- права ---- select agency_id into v_agency from projects where id = p_project_id; if v_agency is null then raise exception 'project not found'; end if; if v_agency <> my_agency() or not can_edit('projects') then raise exception 'forbidden'; end if; if p_end < p_start then raise exception 'end before start'; end if; -- ---- старый период договора (для архива) ---- select start_date, end_date into v_old_start, v_old_end from project_contracts where project_id = p_project_id; -- ---- номер следующего цикла ---- select coalesce(max(cycle_no),0)+1 into v_next from project_cycles where project_id = p_project_id; -- ---- снимок текущего цикла ---- v_snap := jsonb_build_object( 'stages', coalesce(( select jsonb_agg(jsonb_build_object('name',s.name,'status',s.status,'idx',s.idx) order by s.idx) from project_stages s where s.project_id = p_project_id), '[]'::jsonb), 'tasks', coalesce(( select jsonb_agg(jsonb_build_object( 'title', t.title, 'status', t.status, 'stage', coalesce(s.name,'—'), 'assignee_id', t.assignee_id, 'time_spent', coalesce(t.time_spent,0), 'due_date', t.due_date)) from project_tasks t left join project_stages s on s.id = t.stage_id where t.project_id = p_project_id), '[]'::jsonb) ); -- ---- архивируем (только если есть что) ---- if (v_snap->'stages') <> '[]'::jsonb or (v_snap->'tasks') <> '[]'::jsonb then insert into project_cycles(project_id, cycle_no, period_start, period_end, snapshot) values (p_project_id, v_next, v_old_start, v_old_end, v_snap); end if; -- ---- аннулируем задачи ---- delete from project_tasks where project_id = p_project_id; -- ---- сбрасываем этапы ---- update project_stages set status='wait' where project_id = p_project_id; select id into v_first from project_stages where project_id = p_project_id order by idx limit 1; if v_first is not null then update project_stages set status='active' where id = v_first; end if; -- ---- новый период договора ---- insert into project_contracts(project_id, start_date, end_date, updated_at) values (p_project_id, p_start, p_end, now()) on conflict (project_id) do update set start_date = excluded.start_date, end_date = excluded.end_date, updated_at = now(); -- ---- проект активен ---- update projects set status='active' where id = p_project_id; return v_next; end; $$; grant execute on function public.renew_project(uuid, date, date) to authenticated, anon;
